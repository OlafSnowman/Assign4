package program;

import java.util.StringJoiner;

public class StringWriter {
  private String writer = "";

  public void write(String string) {
    writer += string;
  }

  public String getWriter() { //Feedback: we are not quite getting the writer here. Let us name appropriately
    return writer;
  }

  public void toLowerCase() {
    writer = writer.toLowerCase();
  }

  public void toUpperCase() {
    writer = writer.toUpperCase();
  }

  public void removeStupid() {
    writer = writer.replaceAll("stupid", "s*****");
  }

  public void removeDuplicates() {
    String[] words = writer.split(" ");

    int uniqueWords = 1;
    for(int i = 1; i < words.length; i++) {
      if(!words[uniqueWords - 1].equals(words[i])) {
        words[uniqueWords++] = words[i];
      }
    }

    StringJoiner tempWriter = new StringJoiner(" ");
    for(int i = 0; i < uniqueWords; i++) {
      tempWriter.add(words[i]);
    }

    writer = tempWriter.toString();
  }

}
